删除一维数组中的重复项

function hasRepeat(ary)
{
    var nary=ary.sort();
    for(var i=0;i<ary.length-1;i++){
    if (ary[i]==ary[i+1]){
        ary.splice(i,1);
        i--;
        }
    }
    return ary;
}

删除二维数组中重复的一维数组

function Distict(arr)
{
    var hash = {};  
    var result = [];  
    for(var i=0; i < arr.length; i++){  
        if(!hash[arr[i]]){  
           result.push(arr[i]);  
           hash[arr[i]] = true;  
        }  
    }
    return result; 
}
